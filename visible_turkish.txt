    if (cb) cb.checked = !!aktif;
    const rWrap = document.getElementById(s+'RampaWrap');
    if (rWrap) rWrap.style.display = aktif ? '' : 'none';
    const hWrap = document.getElementById(s+'HedefWrap');

      const oranWrap = document.getElementById(s+'OranWrap');
      if (useIst && oranWrap) {
        oranWrap.style.display = '';
        const oran = V[s+'RampaOran'] || 0.4;
        const sl = document.getElementById('s_'+s+'RampaOran');
        if (sl) sl.value = oran;
        const vl = document.getElementById(s+'RampaOranVal');
        if (vl) vl.textContent = '×'+parseFloat(oran).toFixed(2);
      }
      const giderCb = document.getElementById(s+'IstGiderToggle');
      if (giderCb) giderCb.checked = V[s+'UseIstGider'] !== false;
      const kurulumCb = document.getElementById(s+'IstKurulumToggle');
      if (kurulumCb) kurulumCb.checked = V[s+'UseIstKurulum'] !== false;
      if (s === 'ankara') {
        const ay = V.ankaraAcilisAy || 3;
        const sl = document.getElementById('s_ankaraAcilisAy');

      }
      initMerkezRampaChart(s);
      renderMerkezRampaChart(s);
    }
  });
});

function switchInnerTab(btn, tabId) {
  _ALL_GIDER_TABS.forEach(function(id) {
    var el = document.getElementById(id);
    if (el) el.style.display = 'none';
  });
  var show = document.getElementById(tabId);
  if (show) show.style.display = '';
  document.querySelectorAll('.inner-tab').forEach(function(b) { b.classList.remove('active'); });
  btn.classList.add('active');
  recalc();
  if (tabId === 'tab-donemsel') {
    setTimeout(function() { if (window.initDonemCanvas) initDonemCanvas(); }, 30);
  }

function tf(n) {
  if (n === undefined || n === null) return '—';
  if (n === 0) return '₺0';
  var abs = Math.abs(Math.round(n));
  return (n < 0 ? '−₺' : '₺') + abs.toLocaleString('tr-TR');
}
function tfShort(n) {
  if (!n) return '—';
  var abs = Math.abs(Math.round(n));
  return (n < 0 ? '−₺' : '₺') + abs.toLocaleString('tr-TR');
}
function ef(n) { if (!n && n !== 0) return '—'; return '€' + Math.round(n).toLocaleString('tr-TR'); }
function pf(n) { return (n || 0).toFixed(1) + '%'; }

function renderPiyasaKpi() {
  const osteoidK  = (window._lastRows||[]).reduce((s,r)=>s+(r.korse||0),0) || 325;
  const osteoidB2B = ((window._lastRows||[]).length ? V.korseB2B||[] : []).reduce((s,v)=>s+(+v||0),0);
  const osteoidToplam = osteoidK + osteoidB2B;
  const osteoidFiyat  = V.korseF_stdR || 30000;
  const rakipToplam = RAKIPLER.reduce((s,r)=>s+gRV(r,'KorseK',r.dK)+gRV(r,'KorseB',r.dB),0);
  const tumToplam   = osteoidToplam + rakipToplam;
  const osteoidPay  = tumToplam > 0 ? Math.round(osteoidToplam/tumToplam*100) : 0;
  const rakipKlinikTop = RAKIPLER.reduce((s,r)=>s+gRV(r,'KorseK',r.dK),0);
  const ortFiyat = rakipKlinikTop > 0
    ? Math.round(RAKIPLER.reduce((s,r)=>s+gRV(r,'KorseK',r.dK)*gRV(r,'FiyatK',r.dFK),0)/rakipKlinikTop) : 0;

  RAKIPLER.forEach(r => {
    const kK=gRV(r,'KorseK',r.dK), kB=gRV(r,'KorseB',r.dB);
    const fK=gRV(r,'FiyatK',r.dFK), fB=gRV(r,'FiyatB',r.dFB);
    const el = id => document.getElementById(id);
    if(el('kpi_'+r.id+'_K')) el('kpi_'+r.id+'_K').textContent = kK;
    if(el('kpi_'+r.id+'_B')) el('kpi_'+r.id+'_B').textContent = kB;
    if(el('kpi_'+r.id+'_T')) el('kpi_'+r.id+'_T').textContent = kK+kB;
    if(el('kpi_'+r.id+'_C')) el('kpi_'+r.id+'_C').textContent = '₺'+Math.round((kK*fK+kB*fB)/1000).toLocaleString('tr-TR')+'K';
  });

  document.getElementById('piyasaKpiGrid').innerHTML = [
    { label:'Osteoid klinik (yıllık)',  val: osteoidK+' adet',       c:'pos' },
    { label:'Osteoid B2B (yıllık)',     val: osteoidB2B+' adet',     c:'neu' },
    { label:'Rakip toplam (yıllık)',    val: rakipToplam+' adet',     c:'neu' },
    { label:'Osteoid piyasa payı',      val: '%'+osteoidPay,          c: osteoidPay>=10?'pos':'neu' },
    { label:'Rakip ort. klinik fiyat',  val: ff(ortFiyat),            c:'neu' },
    { label:'Osteoid baz fiyat',        val: ff(osteoidFiyat),        c: osteoidFiyat>=ortFiyat?'pos':'neg' },
  ].map(k=>`<div class="kpi"><div class="kpi-label">${k.label}</div><div class="kpi-val ${k.c}">${k.val}</div></div>`).join('');
}

// ── Yardımcı formatters ────────────────────────────────────────────────────────
function tf(n) {
  if (n === undefined || n === null) return '—';
  if (n === 0) return '₺0';
  const abs = Math.abs(Math.round(n));
  return (n < 0 ? '−₺' : '₺') + abs.toLocaleString('tr-TR');
}
function tfShort(n) {
  if (!n) return '—';
  const abs = Math.abs(Math.round(n));
  return (n < 0 ? '−₺' : '₺') + abs.toLocaleString('tr-TR');
}
function ef(n) {
  if (!n && n !== 0) return '—';
  return '€' + Math.round(n).toLocaleString('tr-TR');
}
function pf(n) { return (n || 0).toFixed(1) + '%'; }

// ── SECTION 6: 3 Yıllık Projeksiyon ─────────────────────────────────────────
function renderProjection() {
  const rows = window._lastRows || [];
  const eurKur = V.eurKur || 50;
  const toEur = v => Math.round(v / eurKur / 1000);
  const lerp = (a,b,t) => Math.round(a + (b-a)*t);

  const y1KorseNet = rows.reduce((s,r)=>s+(r.gelirNet||0), 0);
  const y1Korse    = rows.reduce((s,r)=>s+(r.korse||0), 0);
  const y1Gider    = Math.abs(rows.reduce((s,r)=>s+(r.gider||0), 0));
  const y1GiderEur = toEur(y1Gider);

  const pazarTR    = gv('pazarTR');
  const pazarIstPct= gv('pazarIstPct') / 100;
  const pazarIst   = Math.round(pazarTR * pazarIstPct);
  const hedefPay   = gv('hedefOsteoidPay') / 100;
  const y5KorseAdet= Math.round(pazarIst * hedefPay);

  const sonR = rows.slice(-3);
  const sonK = sonR.reduce((s,r)=>s+(r.korse||0),0)||1;
  const sonN = sonR.reduce((s,r)=>s+(r.gelirNet||0),0);
  const birimNetEur = sonN / sonK / eurKur;
  const y5KorseToplam = Math.round(y5KorseAdet * birimNetEur / 1000);

  const y3BrutGelir = y5KorseToplam;
  const y2GiderEur  = Math.round(y1GiderEur * 1.15);
  const y3GiderEur  = Math.round(y1GiderEur * 1.25);
  const y1BrutM1    = toEur(y1KorseNet);
  const y1NetM1     = Math.max(0, y1BrutM1 - y1GiderEur);
  const y2NetM1     = Math.max(0, lerp(y1BrutM1, y3BrutGelir, 0.4) - y2GiderEur);
  const y3NetM1     = Math.max(0, y3BrutGelir - y3GiderEur);
  const y3SgkNet    = Math.round(y3BrutGelir * 0.35 * 0.85);
  const y2SgkNet    = Math.round(y3SgkNet * 0.25);

  const b2bY1 = toEur(window._lastGelirB2B || 0);
  const b2bRow = [b2bY1, Math.round(b2bY1*1.5), Math.round(b2bY1*2.2)];

  const {triggerAy: _trigAy} = calc4AyOrtalama(rows);
  const _acilisAy = _trigAy ? Math.max(13, _trigAy+3) : 13;
  const _aktifAy  = Math.max(0, Math.min(12, 25 - _acilisAy));

  const izmirGelirY2  = V.izmirAktif  ? Math.round(getMerkezGelir('izmir')  * _aktifAy / 12) : 0;
  const ankaraGelirY2 = V.ankaraAktif ? Math.round(getMerkezGelir('ankara') * _aktifAy / 12) : 0;
  const pazarIzmir  = Math.round(gv('pazarTR') * V.izmirNufusPay / 100);
  const pazarAnkara = Math.round(gv('pazarTR') * V.ankaraNufusPay / 100);
  const y1BirimNetEur = toEur(y1KorseNet) / (y1Korse||1) * 1000;
  const izmirY3Adet   = V.izmirAktif  ? Math.round(pazarIzmir  * gv('izmirHedefPay')  / 100) : 0;
  const ankaraY3Adet  = V.ankaraAktif ? Math.round(pazarAnkara * gv('ankaraHedefPay') / 100) : 0;
  const izmirY3Gelir  = Math.round(izmirY3Adet  * y1BirimNetEur / 1000);
  const ankaraY3Gelir = Math.round(ankaraY3Adet * y1BirimNetEur / 1000);

  const sonBirimNet = Math.round(sonN / sonK);

  document.getElementById('formulProj').innerHTML =
    '<span class="c">── YIL 1 ───────────────────────────────────────────────────────────────────</span>\n' +
    '<span class="k">Yıl1_M1_Net</span>    = 12 aylık modelden direkt → €'+y1NetM1+'K\n' +
    '<span class="k">Yıl1_GiderEur</span>  = -(Σ aylık gider) / eurKur / 1000 → €'+y1GiderEur+'K\n\n' +
    '<span class="c">── YIL 2 ───────────────────────────────────────────────────────────────────</span>\n' +
    '<span class="k">Yıl2_BrutM1</span>    = DOĞRUSAL(Yıl1=€'+y1BrutM1+'K → Yıl3=€'+y3BrutGelir+'K, t=0.4)\n' +
    '<span class="k">Yıl2_GiderEur</span>  = Yıl1_Gider × 1.15 (personel + kira artışı) = €'+y2GiderEur+'K\n' +
    '<span class="k">Yıl2_M1_Net</span>    = MAKS(0, Yıl2_Brut − Yıl2_Gider) → €'+y2NetM1+'K\n' +
    '<span class="k">Yıl2_SGK_Net</span>   = Yıl3_SGK × 0.25 (ramp-up %25) → €'+y2SgkNet+'K\n\n' +
    '<span class="c">── YIL 3 ───────────────────────────────────────────────────────────────────</span>\n' +
    '<span class="k">Yıl3_Hedef_Korse</span>= PazarIST × %'+gv('hedefOsteoidPay')+' = '+pazarIst.toLocaleString('tr-TR')+' × %'+gv('hedefOsteoidPay')+' = '+y5KorseAdet+' adet\n' +
    '<span class="k">BirimNetEur</span>     = Yılsonu ortalama (son 3 ay) = ₺'+sonBirimNet.toLocaleString('tr-TR')+'/korse\n' +
    '<span class="k">Yıl3_BrutGelir</span> = '+y5KorseAdet+' × ₺'+sonBirimNet.toLocaleString('tr-TR')+' / '+eurKur+' / 1000 = €'+y3BrutGelir+'K\n' +
    '<span class="k">Yıl3_GiderEur</span>  = Yıl1_Gider × 1.25 → €'+y3GiderEur+'K\n' +
    '<span class="r">Yıl3_M1_Net</span>    = MAKS(0, Yıl3_Brut − Yıl3_Gider) → €'+y3NetM1+'K\n' +
    '<span class="k">Yıl3_SGK_Net</span>   = Yıl3_M1_Net × 0.35 × 0.85 → €'+y3SgkNet+'K\n' +
    '<span class="c">            0.35 = SGK kanal oranı öngörüsü · 0.85 = tahsilat etkinliği (60–90 gün gecikme)</span>';

  // Projeksiyon tablosu
  const fmt = v => v>0 ? '€'+v+'K' : '—';
  const totals = [
    y1NetM1 + 0 + 0 + b2bRow[0] + (V.izmirAktif?0:0) + (V.ankaraAktif?0:0),
    y2NetM1 + y2SgkNet + 0 + b2bRow[1] + izmirGelirY2 + ankaraGelirY2,
    y3NetM1 + y3SgkNet + 0 + b2bRow[2] + izmirY3Gelir + ankaraY3Gelir,
  ];

  const trows = [
    ['Korse — M1 (özel kanal)', 'Model Y1; büyüme projeksiyon Y2-3', y1NetM1, y2NetM1, y3NetM1],
    ['Korse — M1 (SGK, Yıl 2+)', 'Y3 × 0.35 × 0.85; Y2 = Y3 × 0.25', 0, y2SgkNet, y3SgkNet],
    ...(b2bRow[0]>0||b2bRow[1]>0 ? [['B2B — İstanbul', 'Y1 modelden; Y2=×1.5; Y3=×2.2', b2bRow[0], b2bRow[1], b2bRow[2]]] : []),
    ...(V.izmirAktif ? [['Korse — İzmir', 'getMerkezGelir × aktifAy/12; Y3: nüfus pay × birimNet', 0, izmirGelirY2, izmirY3Gelir]] : []),
    ...(V.ankaraAktif ? [['Korse — Ankara', 'getMerkezGelir × aktifAy/12; Y3: nüfus pay × birimNet', 0, ankaraGelirY2, ankaraY3Gelir]] : []),
  ];

  document.getElementById('projTbody').innerHTML =
    trows.map(([name, formula, y1, y2, y3]) =>
      '<tr><td>'+name+'</td><td class="formula">'+formula+'</td>' +
      '<td class="result">'+(y1?fmt(y1):'—')+'</td>' +
      '<td class="result">'+(y2?fmt(y2):'—')+'</td>' +
      '<td class="result">'+(y3?fmt(y3):'—')+'</td></tr>'
    ).join('') +
    '<tr class="grp"><td>Konsolide Toplam</td><td class="formula">Σ tüm kalemler</td>' +
    '<td class="result">€'+totals[0]+'K</td><td class="result">€'+totals[1]+'K</td><td class="result">€'+totals[2]+'K</td></tr>';
}

Ciro = (Yıl 5 FAVÖK + olgunluk dönemi yıllık opex) / ürün brüt marjı. Bu formül, FAVÖK'ün opex indirildikten sonraki kâr olduğu gerçeğinden hareket eder: FAVÖK = Ciro × marj − opex → Ciro = (FAVÖK + opex) / marj. Ürün brüt marjı (hekim komisyonu ve malzeme sonrası, opex öncesi) Yıl 1 verilerinden türetilir ve büyüme ile sabit kabul edilir.
FAVÖK (EBITDA) = Vergi, faiz ve amortisman öncesi kâr. Bu modelde faiz ve amortisman kalemi olmadığından FAVÖK ≈ işletme kârı (FVÖK). KDV: ortez-protez teslimi KDV Kanunu 17/4-s kapsamında %0 (tam istisna). Kurumlar vergisi: klinik faaliyetinden elde edilen kâr üzeri %25; satış istisnası yoktur (TGB muafiyeti yalnızca Osteoid A.Ş.'ye aittir). EV her zaman FAVÖK üzerinden hesaplanır — yukarıdaki rakamlar hisse değeri değil toplam işletme değeridir. Hisse değeriniz = EV × hisse oranınız. const GDOC_KEY = 'osteoid_gdoc';

function saveGdocUrl() {
  var url = document.getElementById('gdocUrl').value.trim();
  localStorage.setItem(GDOC_KEY, url);
  var st = document.getElementById('gdocStatus');
  st.textContent = url ? '✓ Kaydedildi' : 'URL temizlendi.';
  st.style.color = url ? '#1a7a45' : '#888';
}

function openGdoc() {
  var url = (document.getElementById('gdocUrl').value.trim()) || localStorage.getItem(GDOC_KEY) || '';
  if (!url) {
    var st = document.getElementById('gdocStatus');
    st.textContent = 'Önce bir Google Docs bağlantısı girin ve kaydedin.';
    st.style.color = '#c0392b';
    return;
  }
  window.open(url, '_blank');
}

function saveTerm(key, el) {
  localStorage.setItem('osteoid_term_' + key, el.textContent.trim());
}

function switchInnerTab(btn, tabId) {
  ['tab-yatirimci','tab-sozlesme'].forEach(function(id) {
    var el = document.getElementById(id); if (el) el.style.display = 'none';
  });
  var show = document.getElementById(tabId); if (show) show.style.display = '';
  document.querySelectorAll('.inner-tab').forEach(function(b) { b.classList.remove('active'); });
  btn.classList.add('active');
  recalc();
}

  recalc();

  var url = localStorage.getItem(GDOC_KEY) || 'https://docs.google.com/document/d/1A6S8b0mF-fuwrObXXbo11ZzjeAXH3XvB5TcYdH1_aao/edit?usp=sharing';
  var gdocEl = document.getElementById('gdocUrl');
  if (gdocEl) gdocEl.value = url;
  if (url) {
    var st = document.getElementById('gdocStatus');
    if (st) { st.textContent = '✓ Kayıtlı bağlantı yüklendi'; st.style.color = '#1a7a45'; }
  }

  var invest = V.dcfInvest || 150000;
  var tutarEl = document.getElementById('term_Y_tutar');
  if (tutarEl) tutarEl.textContent = '€' + Math.round(invest).toLocaleString('tr-TR');

  var komisyonEl = document.getElementById('term_H_komisyon');
  if (komisyonEl) komisyonEl.textContent = '%' + (V.danisPct || 25);

  ['Y_arac','Y_kontrol','Y_temettu','Y_raporlama','Y_onalim','Y_cikis','Y_rekabet','Y_teknoloji',
   'H_odeme','H_fatura','H_kanal','H_sure','H_gizlilik','H_fesih','H_rekabet'].forEach(function(k) {

  });
});
